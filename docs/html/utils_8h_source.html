265 class MidiHook : public QObject {
266  Q_OBJECT
267 public:
268  int channel = -1; //midi channel
269  QString message_type; // Message Type
270  int norc = -1; // Note or Control
271  QString action;
272  QString scene;
273  QString source;
274  QString filter;
275  QString transition;
276  QString item;
277  QString audio_source;
278  QString media_source;
279  int duration = -1;
280  QString scene_collection;
281  QString profile;
282  QString string_override;
283  bool bool_override = false;
284  int int_override = -1;
285  int value = -1;
286  MidiMessage *get_message_from_hook() const
287  {
288  MidiMessage *message = new MidiMessage();
289  message->channel = this->channel;
290  message->message_type = this->message_type;
291  message->NORC = this->norc;
292  return message;
293  }
294  MidiHook(){};
295  MidiHook(const QString &jsonString)
296  {
297  obs_data_t *data = obs_data_create_from_json(jsonString.toStdString().c_str());
298  channel = obs_data_get_int(data, "channel");
299  message_type = obs_data_get_string(data, "message_type");
300  norc = obs_data_get_int(data, "norc");
301  action = obs_data_get_string(data, "action");
302  scene = obs_data_get_string(data, "scene");
303  source = obs_data_get_string(data, "source");
304  filter = obs_data_get_string(data, "filter");
305  transition = obs_data_get_string(data, "transition");
306  item = obs_data_get_string(data, "item");
307  audio_source = obs_data_get_string(data, "audio_source");
308  media_source = obs_data_get_string(data, "media_source");
309  duration = obs_data_get_int(data, "duration");
310  scene_collection = obs_data_get_string(data, "scene_collection");
311  profile = obs_data_get_string(data, "profile");
312  string_override = obs_data_get_string(data, "string_override");
313  bool_override = obs_data_get_bool(data, "bool_override");
314  int_override = obs_data_get_int(data, "int_override");
315  }
316  obs_data_t *GetData() const
317  {
318  obs_data_t *data = obs_data_create();
319  obs_data_set_int(data, "channel", channel);
320  obs_data_set_string(data, "message_type", message_type.toStdString().c_str());
321  obs_data_set_int(data, "norc", norc);
322  obs_data_set_string(data, "action", action.toStdString().c_str());
323  obs_data_set_string(data, "scene", scene.toStdString().c_str());
324  obs_data_set_string(data, "source", source.toStdString().c_str());
325  obs_data_set_string(data, "filter", filter.toStdString().c_str());
326  obs_data_set_string(data, "transition", transition.toStdString().c_str());
327  obs_data_set_string(data, "item", item.toStdString().c_str());
328  obs_data_set_string(data, "audio_source", audio_source.toStdString().c_str());
329  obs_data_set_string(data, "media_source", media_source.toStdString().c_str());
330  obs_data_set_int(data, "duration", duration);
331  obs_data_set_string(data, "scene_collection", scene_collection.toStdString().c_str());
332  obs_data_set_string(data, "profile", profile.toStdString().c_str());
333  obs_data_set_string(data, "string_override", string_override.toStdString().c_str());
334  obs_data_set_bool(data, "bool_override", bool_override);
335  obs_data_set_int(data, "int_override", int_override);
336  return data;
337  }
338  const char *ToJSON() { return obs_data_get_json(GetData()); }
339 };
